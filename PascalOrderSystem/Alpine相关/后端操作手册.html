后端操作手册

本手册记录了从零开始配置服务器、数据库、后端API、前端页面，并完成数据提交与验证的全过程。所有命令均基于 Alpine Linux 系统（MariaDB 11.4.11, PHP 8.4.21）。请根据您的实际环境调整路径、IP和密码。

一、服务器环境准备

1.1 确认 MariaDB 监听地址（允许远程连接）

netstat -tlnp | grep 3306

期望输出包含 0.0.0.0:3306。若只监听 127.0.0.1，需修改配置文件：

vi /etc/my.cnf.d/mariadb-server.cnf
# 在 [mysqld] 部分添加或修改：
bind-address = 0.0.0.0

重启 MariaDB：

rc-service mariadb restart

1.2 登录 MariaDB 并创建数据库及用户

mariadb -u root -p --skip-ssl   # 如果root无密码，直接回车

在 MariaDB 提示符下依次执行：

-- 创建数据库（使用 utf8mb4 字符集）
CREATE DATABASE IF NOT EXISTS InvStrat_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 创建远程用户（允许任何IP连接，密码请修改）
CREATE USER 'jefflu'@'%' IDENTIFIED BY '780503';

-- 授权
GRANT ALL PRIVILEGES ON InvStrat_db.* TO 'jefflu'@'%';

-- 刷新权限
FLUSH PRIVILEGES;

-- 退出
EXIT;

1.3 创建数据表（三张表）

mariadb -u jefflu -p --skip-ssl   # 密码 780503
USE InvStrat_db;

CREATE TABLE geo_env_events (
    id BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT UNSIGNED NOT NULL,
    event_time DATETIME NOT NULL,
    title VARCHAR(255) NOT NULL,
    source VARCHAR(100) DEFAULT NULL,
    region VARCHAR(100) DEFAULT NULL,
    country_involved VARCHAR(255) DEFAULT NULL,
    event_type ENUM('conflict','diplomacy','health','economy','security','other') NOT NULL DEFAULT 'other',
    severity_level TINYINT UNSIGNED NOT NULL DEFAULT 1,
    description TEXT DEFAULT NULL,
    impact_summary TEXT DEFAULT NULL,
    risk_score DECIMAL(5,2) DEFAULT NULL,
    tags VARCHAR(255) DEFAULT NULL,
    raw_content MEDIUMTEXT DEFAULT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE geo_impact_assessments (
    id BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    event_id BIGINT UNSIGNED NOT NULL,
    asset_class ENUM('oil','stock','currency','commodity','bond','other') DEFAULT NULL,
    industry VARCHAR(100) DEFAULT NULL,
    impact_direction ENUM('positive','negative','neutral','mixed') DEFAULT NULL,
    impact_level TINYINT UNSIGNED DEFAULT NULL,
    expected_duration ENUM('short','medium','long') DEFAULT NULL,
    description TEXT DEFAULT NULL,
    FOREIGN KEY (event_id) REFERENCES geo_env_events(id) ON DELETE CASCADE
);

CREATE TABLE geo_event_sources (
    id BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    event_id BIGINT UNSIGNED NOT NULL,
    source_name VARCHAR(100) DEFAULT NULL,
    source_url VARCHAR(500) DEFAULT NULL,
    author VARCHAR(100) DEFAULT NULL,
    publish_time DATETIME DEFAULT NULL,
    excerpt TEXT DEFAULT NULL,
    FOREIGN KEY (event_id) REFERENCES geo_env_events(id) ON DELETE CASCADE
);

EXIT;

二、后端 API 搭建

2.1 安装 PHP 及扩展

apk update
apk add php php-pdo_mysql php-json php-common
php -v   # 验证安装

2.2 创建工作目录和 API 文件

mkdir -p /home/jefflu/api
cd /home/jefflu/api

创建 submit.php（直接使用 cat 命令写入，避免格式错误）：

cat > submit.php << 'EOF'
setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);
    $pdo->beginTransaction();

    $event = $input['event'];
    $sqlEvent = "INSERT INTO geo_env_events 
        (user_id, event_time, title, source, region, country_involved, event_type, 
         severity_level, description, impact_summary, risk_score, tags, raw_content, created_at)
        VALUES 
        (:user_id, :event_time, :title, :source, :region, :country_involved, :event_type,
         :severity_level, :description, :impact_summary, :risk_score, :tags, :raw_content, NOW())";
    $stmt = $pdo->prepare($sqlEvent);
    $stmt->execute([
        ':user_id' => $event['user_id'] ?? 1,
        ':event_time' => $event['event_time'],
        ':title' => $event['title'],
        ':source' => $event['source'] ?? null,
        ':region' => $event['region'] ?? null,
        ':country_involved' => is_array($event['country_involved']) ? implode(',', $event['country_involved']) : $event['country_involved'],
        ':event_type' => $event['event_type'],
        ':severity_level' => $event['severity_level'] ?? 3,
        ':description' => $event['description'] ?? null,
        ':impact_summary' => $event['impact_summary'] ?? null,
        ':risk_score' => $event['risk_score'] ?? null,
        ':tags' => is_array($event['tags']) ? implode(',', $event['tags']) : $event['tags'],
        ':raw_content' => $event['raw_content'] ?? null
    ]);
    $eventId = $pdo->lastInsertId();

    if (!empty($input['impact_assessments'])) {
        $sqlImpact = "INSERT INTO geo_impact_assessments 
            (event_id, asset_class, industry, impact_direction, impact_level, expected_duration, description)
            VALUES (?, ?, ?, ?, ?, ?, ?)";
        $stmtImpact = $pdo->prepare($sqlImpact);
        foreach ($input['impact_assessments'] as $imp) {
            $stmtImpact->execute([
                $eventId,
                $imp['asset_class'] ?? null,
                $imp['industry'] ?? null,
                $imp['impact_direction'] ?? null,
                $imp['impact_level'] ?? 3,
                $imp['expected_duration'] ?? 'short',
                $imp['description'] ?? null
            ]);
        }
    }

    if (!empty($input['sources'])) {
        $sqlSource = "INSERT INTO geo_event_sources 
            (event_id, source_name, source_url, author, publish_time, excerpt)
            VALUES (?, ?, ?, ?, ?, ?)";
        $stmtSource = $pdo->prepare($sqlSource);
        foreach ($input['sources'] as $src) {
            $stmtSource->execute([
                $eventId,
                $src['source_name'] ?? null,
                $src['source_url'] ?? null,
                $src['author'] ?? null,
                $src['publish_time'] ?? null,
                $src['excerpt'] ?? null
            ]);
        }
    }

    $pdo->commit();
    echo json_encode(['success' => true, 'event_id' => $eventId]);
} catch (Exception $e) {
    if (isset($pdo)) $pdo->rollBack();
    http_response_code(500);
    echo json_encode(['success' => false, 'error' => $e->getMessage()]);
}
EOF

2.3 启动 PHP 内置服务器

六、常见问题及解决

问题解决方法
连接数据库时 SSL required在 mysql 命令后加 --skip-ssl
前端提交返回 Unexpected token检查 submit.php 开头是否有 <?php，且无 BOM 头；确保文件未被输出额外内容
提交后出现约束错误 CONSTRAINT ... tags failed删除旧表重建（参考 1.3 节）
跨域错误（CORS）确保 submit.php 中已处理 OPTIONS 请求且设置了 Access-Control-Allow-Origin: *
PHP 服务器启动后无法访问检查防火墙（Alpine 默认无防火墙），确保端口 8000 开放；使用 curl http://127.0.0.1:8000/submit.php 测试

创建 OpenRC 服务，PHP开机自启动

# 打开脚本文件
vi /etc/init.d/geoen-php

以下是文件内容：

#!/sbin/openrc-run

name="GeoEnv PHP Service"
description="PHP Built-in Server for GeoEnv Application"

command="/usr/bin/php"
command_args="-S 0.0.0.0:8000 -t /home/jefflu/api"
command_user="jefflu"
pidfile="/run/${RC_SVCNAME}.pid"
command_background=true

depend() {
    need net
    after firewall
}

按 i 进入编辑模式,按 Esc，输入 :wq 保存退出。

设置权限并加入开机自启

chmod +x /etc/init.d/geoen-php
rc-update add geoen-php default
rc-service geoen-php start

管理服务

rc-service geoen-php start      # 启动
rc-service geoen-php stop       # 停止
rc-service geoen-php restart    # 重启
rc-service geoen-php status     # 查看状态
